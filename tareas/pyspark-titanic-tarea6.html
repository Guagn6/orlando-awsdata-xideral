# Crear Sesin
spark = SparkSession.builder \
    .appName("analisis_titanic") \
    .master("local[*]") \
    .config("spark.executor.memory","4g") \
    .config("spark.driver.memory","2g") \
    .config("spark.executor.cores", "2") \
    .config("spark.serializer", "org.apache.spark.serializer.KryoSerializer") \
    .config("spark.sql.files.maxPartitionBytes", "128MB") \
    .config("spark.sql.shuffle.partitions", "200") \
    .config("spark.sql.execution.arrow.enabled", "true") \
    .getOrCreate()

# Tamao de Datos
# Nmero de particiones = (Tamao Total de Datos) / (Tamao Ideal de Particion)
# 10gb (Tamao Total) / 128mb = 80 particiones

# SSD HDFS ms particiones
# HDD, RED Menos particiones
# Particiones de:
    # < 50mb sobrecarga en la administracin
    # > 50mb problemas de memoria y cuellos de botella 
# Mxima cantidad de operaciones en paralelo = Nmero de ncleos X 2
# Tamao ptimo de la particin para la Lectura de Datos = Tamao de particin / Nmero de particiones
