import json
import boto3
import io
import pandas as pd

s3=boto3.client("s3")

def lambda_handler(event, context):
    # Descargar archivo de S3 a memoria
    buffer = io.BytesIO()
    s3.download_fileobj(
        Bucket = "xideralaws-curso-orlando",
        Key = "nyc_taxi_2023/yellow_tripdata_2023-01.parquet",
        Fileobj = buffer)
    
    # Cargar parquet en pandas
    buffer.seek(0) # Regresa puntero al inicio
    df = pd.read_parquet(buffer, engine="pyarrow")

    # Limpiar datos
    df = df.dropna()
    df = df.drop_duplicates()

    # Preview de 2 registros    
    preview = df.head(2).to_dict(orient="records")

    return {
        'statusCode': 200,
        'body': json.dumps(preview, default=str)
    }
