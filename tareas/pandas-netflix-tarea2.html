Promedio de Duracion por TV Shows y Peliculas: Método 1¶

In [8]:
# str.replace(): Se encadena a una Series tipo text; reemplaza coincidencias de un patrón (string o regex).
# Argumentos: 1. pat (qué se reemplaza, str o regex), 2. repl (con qué se reemplaza str o callable) y
# 3. regex (bool, default=None), si es True, interpreta a pat como expresioón regular. Retorna una Series con textos modificados.
df["duration"] = df["duration"].str.replace(r'[^0-9]', '', regex=True).astype(float)

# .loc[]: Tiene varios usos para selecciona y asignar datos dentro de un DataFrame por filas y/o columnas:
    # 1. Seleccionar filas por condición --> df.loc[df["duration"] > 100]   
    # 2. Seleccionar filas y columnas:
            #df.loc[0, "duration"]             * Valor en la fila 0, columna duration
            #df.loc[:, ["duration", "title"]]  * Todas las filas, solo columnas duration y title
    # 3. Asignar valores: --> df.loc[df["duration"] > 200, "category"] = "Long" - Aquí se cambia "category" por "Long" 
    # en filas donde "duration" es mayor a 200.
avgMovie = df.loc[df["type"] == "Movie", "duration"].mean()
avgTvShow = df.loc[df["type"] == "TV Show", "duration"].mean()
